async function fetchData() { const searchInput = document.getElementById('searchInput'); const search = searchInput ? searchInput.value : ''; const display = document.getElementById('display-area'); if (!display) return; // 表示エリアがない場合は中断 display.innerHTML = '
読み込み中...
'; try { // Supabaseからデータを取得 let query = supabaseClient.from('items').select('*'); if (search) { query = query.ilike('name', `%${search}%`); } const { data, error } = await query; if (error) throw error; // データの表示処理 display.innerHTML = ''; if (data.length === 0) { display.innerHTML = '
該当するデータが見つかりませんでした。
'; return; } data.forEach(item => { const card = document.createElement('div'); card.className = 'card'; // HTMLの中身を構築 card.innerHTML = `
${item.category || ''}
${item.name || '名称未設定'}
📍 ${item.location || '不明'}

${item.description || '説明なし'}

🛒 買値: ${item.buy_price || '-'} 🪙 売値: ${item.sell_price || '-'}
`; display.appendChild(card); }); } catch (err) { console.error('エラーの詳細:', err); display.innerHTML = `
エラーが発生しました: ${err.message}
`; } }
